async function fetchLog() { const logDiv = document.getElementById('matchLog'); try { // Fetch matches const mRes = await fetch('matches.json?cb=' + Date.now()); const games = await mRes.json(); // Fetch halftime (with a silent fail if it doesn't exist) let htScores = []; try { const hRes = await fetch('halftime.json?cb=' + Date.now()); if (hRes.ok) htScores = await hRes.json(); } catch (e) { console.warn("Halftime data not found, continuing without it."); } // Create lookup map const htMap = {}; htScores.forEach(h => { // Using a simple key for matching const key = `${h.home}-${h.away}`.toLowerCase().replace(/\s/g, ''); htMap[key] = `${h.ht_home}-${h.ht_away}`; }); logDiv.innerHTML = games.map(g => { const home = g.home || g.team1; const away = g.away || g.team2; const key = `${home}-${away}`.toLowerCase().replace(/\s/g, ''); const htScore = htMap[key] || '-'; // Shows '-' if no halftime data return `
${home} v ${away}
${owner(home)} / ${owner(away)}
${htScore}
${g.score1 || 0}-${g.score2 || 0}
8
0
`; }).join(''); } catch(e) { logDiv.innerHTML = "Error loading data. Check console (F12) for details."; console.error("Critical Error:", e); } }
